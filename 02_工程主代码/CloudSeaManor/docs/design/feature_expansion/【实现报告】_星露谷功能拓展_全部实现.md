## 五、集成指南

### 5.1 系统初始化

在 `GameRuntime::Initialize()` 中添加：

```cpp
#include "CloudSeamanor/FishingSystem.hpp"
#include "CloudSeamanor/MysticMirrorSystem.hpp"
#include "CloudSeamanor/ContractSystem.hpp"

// 初始化钓鱼系统
CloudSeamanor::domain::FishingSystem::Instance().Initialize();

// 初始化观云镜系统
CloudSeamanor::domain::MysticMirrorSystem::Instance().Initialize();

// 初始化契约系统
CloudSeamanor::domain::ContractSystem::Instance().Initialize();
```

### 5.2 UI管理器初始化

在 `GameRuntime::Initialize()` 中添加：

```cpp
#include "CloudSeamanor/engine/FishingMiniGame.hpp"
#include "CloudSeamanor/engine/PixelMysticMirrorPanel.hpp"
#include "CloudSeamanor/engine/PixelContractPanel.hpp"

// 初始化UI
CloudSeamanor::engine::FishingUI::Instance().Initialize();
CloudSeamanor::engine::MysticMirrorUI::Instance().Initialize();
CloudSeamanor::engine::ContractUI::Instance().Initialize();
```

### 5.3 输入处理

在 `GameRuntime::HandleInput()` 中添加：

```cpp
// 钓鱼小游戏输入
if (FishingUI::Instance().IsFishing()) {
    FishingUI::Instance().HandleInput(event);
    return true;
}

// 观云镜输入
if (MysticMirrorUI::Instance().IsVisible()) {
    if (MysticMirrorUI::Instance().HandleInput(event)) {
        return true;
    }
}

// 契约面板输入
if (ContractUI::Instance().IsVisible()) {
    if (ContractUI::Instance().HandleInput(event)) {
        return true;
    }
}
```

### 5.4 渲染处理

在 `GameRuntime::Render()` 中添加：

```cpp
// 钓鱼小游戏渲染
FishingUI::Instance().Render(target);

// 观云镜渲染
MysticMirrorUI::Instance().Render(target);

// 契约面板渲染
ContractUI::Instance().Render(target);
```

### 5.5 触发交互

```cpp
// 钓鱼（在水边按E）
void PlayerInteractRuntime::TryFishing() {
    if (IsNearWater()) {
        auto& fishing = FishingUI::Instance();
        auto& system = CloudSeamanor::domain::FishingSystem::Instance();

        // 获取当前地点和难度
        int difficulty = system.CalculateFinalDifficulty(1, current_location_);
        fishing.StartFishing(difficulty);
    }
}

// 观云镜（主屋内按E）
void PlayerInteractRuntime::TryMysticMirror() {
    if (IsNearMirror()) {
        MysticMirrorUI::Instance().TriggerInteraction();
    }
}

// 契约面板（茶室内按E）
void PlayerInteractRuntime::TryContract() {
    if (IsNearTeaRoom()) {
        ContractUI::Instance().TriggerInteraction();
    }
}
```
