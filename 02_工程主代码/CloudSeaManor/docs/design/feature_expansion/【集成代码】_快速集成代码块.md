void GameApp::TryStartFishing() {
    // 获取当前地图ID
    const std::string& map_id = runtime_.WorldState().GetCurrentMapId();

    // 确定钓鱼地点
    std::string location_id = "main_farm";
    if (map_id.find("river") != std::string::npos) {
        location_id = "river";
    } else if (map_id.find("spirit_realm") != std::string::npos) {
        location_id = "spirit_realm";
    }

    // 计算难度
    auto& fishing_system = CloudSeamanor::domain::FishingSystem::Instance();
    int difficulty = fishing_system.CalculateFinalDifficulty(1, location_id);

    // 开始钓鱼
    auto& fishing_ui = CloudSeamanor::engine::FishingUI::Instance();
    fishing_ui.StartFishing(difficulty);

    // 设置回调
    fishing_ui.SetOnCatchCallback([this](bool success, const std::string& fish_id) {
        if (success && !fish_id.empty()) {
            // 将鱼添加到背包
            runtime_.WorldState().MutablePlayer().AddToInventory(fish_id, 1);

            // 获取鱼名
            const auto* fish_data = CloudSeamanor::domain::FishingSystem::Instance().GetFishData(fish_id);
            std::string fish_name = fish_data ? fish_data->name_zh : fish_id;

            // 显示提示
            CloudSeamanor::engine::SetHintMessage("钓到了 " + fish_name + "！");
        }
    });
}
