### 2.4 核心逻辑改动

#### 2.4.1 `CropGrowthSystem.cpp` — Update() 扩展

```cpp
// 修改现有 Update() 逻辑，追加以下逻辑分支：

void CropGrowthSystem::Update(GameWorldState& world_state, float delta_seconds) {
    for (auto& plot : world_state.MutableTeaPlots()) {
        if (!plot.cleared) continue;
        if (!plot.seeded || (!plot.watered && !plot.sprinkler_installed)) continue;
        if (plot.ready) continue;

        // === 追加1：夜晚生长减速 ===
        float time_modifier = IsDaytime(world_state.clock) ? 1.0f : 0.5f;

        // === 追加2：云海生长加速（替换现有hardcode）===
        float cloud_modifier = GetCloudGrowthModifier(cloud_system.CurrentState());
        //  Clear=1.0, Mist=1.15, DenseCloud=1.40, Tide=1.60

        // === 追加3：施肥加成（使用 CropDefinition.fertilizer_multiplier）===
        float fertilizer_bonus = 0.0f;
        if (plot.fertilized) {
            fertilizer_bonus = (plot.fertilizer_type == "premium") ? 0.45f
                           : (plot.fertilizer_type == "spirit") ? 0.60f
                           : 0.20f;
            // 灵茶肥料效果更好
            if (crop_def.is_spirit_tea) {
                fertilizer_bonus *= crop_def.fertilizer_multiplier;
            }
        }

        float growth_multiplier = 1.0f + fertilizer_bonus;
        if (plot.in_greenhouse) growth_multiplier *= 1.1f;
        if (plot.sprinkler_installed) growth_multiplier *= 1.05f;

        // === 追加4：品质快照记录 ===
        if (plot.stage == 0 && !plot.seed_item_id.empty()) {
            plot.cloud_density_at_planting = cloud_system.CurrentSpiritDensity();
            plot.aura_at_planting = cloud_system.SpiritEnergy();
        }

        // === 追加5：灵化变种判定 ===
        if (cloud_system.IsDenseOrTide() && !plot.spirit_mutated && plot.ready) {
            float mutate_chance = cloud_system.IsTide() ? 0.30f : 0.15f;
            if (Random01() < mutate_chance) {
                plot.spirit_mutated = true;
                plot.harvest_amount = static_cast<int>(plot.harvest_amount * 2.0f);
                // 触发灵化视觉效果
            }
        }

        plot.growth += delta_seconds * growth_multiplier * cloud_modifier * time_modifier;

        // 阶段更新（保持现有逻辑）
        int new_stage = static_cast<int>(
            plot.growth / plot.growth_time * plot.growth_stages);
        new_stage = std::min(new_stage, plot.growth_stages);
        if (new_stage > plot.stage) {
            plot.stage = new_stage;
            refresh_plot_visual(plot);
        }

        if (plot.stage >= plot.growth_stages) {
            plot.ready = true;
            // === 追加6：品质最终判定 ===
            plot.quality = CalculateQualityFromSnapshot(plot, cloud_system);
        }
    }

    // === 新增：TeaBush 采摘蓄力更新 ===
    for (auto& bush : world_state.MutableTeaBushes()) {
        if (!bush.ready_to_harvest) {
            bush.harvest_progress += delta_seconds / (bush.harvest_interval_days * 1440.0f);
            if (bush.harvest_progress >= 1.0f) {
                bush.ready_to_harvest = true;
            }
        }
    }
}
```
